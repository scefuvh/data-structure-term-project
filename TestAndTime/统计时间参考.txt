#include <iostream>
#include <time.h>
#include <Windows.h>
using namespace std;

void move(int count, int start, int finish, int temp)

{

   if (count > 0) {

      move(count - 1, start, temp, finish);

      //cout << "Move disk " << count << " from " << start//统计时间的时候把输出语句注释

       //    << " to " << finish << "." << endl;

      move(count - 1, temp, finish, start);

   }

}

void move2(int count, int start, int finish, int temp)
{
	int swap;
	while (count > 0) {
		move(count - 1, start, temp, finish);
		//cout << "Move disk " << count << " from " << start
		//<< " to " << finish << "." << endl;
		count--;
		swap = start;
		start = temp;
		temp = swap;
	}
}

int main()
{
   LARGE_INTEGER begin;
   LARGE_INTEGER end;
   LARGE_INTEGER frequ;
   int i;
   for(i=26;i<=26; i++){
   cout<<i<<" disks:"<<endl;
   QueryPerformanceFrequency(&frequ);
   QueryPerformanceCounter(&begin); //获取开始的时刻
   move(i, 1, 3, 2);
   QueryPerformanceCounter(&end);//获取结束的时刻
   //计算二者的差值即为运行时间。

   cout<<fixed<<"move :"<<(end.QuadPart - begin.QuadPart)/(double)frequ.QuadPart<<endl;

   QueryPerformanceFrequency(&frequ);
   QueryPerformanceCounter(&begin); //获取开始的时刻
   move2(i, 1, 3, 2);
   QueryPerformanceCounter(&end);//获取结束的时刻
   //计算二者的差值即为运行时间。
   cout<<"move2:"<<(end.QuadPart - begin.QuadPart)/(double)frequ.QuadPart<<endl<<endl;

   }
   return 0;

}
